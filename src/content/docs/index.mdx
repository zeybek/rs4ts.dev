export const tsCode = `type User = { id: string; name: string };

async function getUser(id: string): Promise<User> {
  const url = "/api/users/" + id;
  const res = await fetch(url);
  if (!res.ok) throw new Error("not found");
  return res.json();
}`;

export const rustCode = `struct User { id: String, name: String }

async fn get_user(id: &str) -> Result<User, Error> {
    let url = format!("/api/users/{id}");
    let res = reqwest::get(url).await?;
    if !res.status().is_success() {
        return Err(Error::NotFound);
    }
    Ok(res.json().await?)
}`;

export const ownCode = `let s = String::from("hello");
let s2 = s;            // ownership moves to s2
// println!("{s}");    // compile error: s no longer valid
println!("{s2}");      // ok — s2 owns the string`;

export const errCode = `fn parse_double(input: &str) -> Result<i32, ParseIntError> {
    let n: i32 = input.trim().parse()?;   // ? propagates the error
    Ok(n * 2)
}`;

export const asyncCode = `#[tokio::main]
async fn main() {
    // both run concurrently — no event loop required
    let (a, b) = tokio::join!(fetch_user(1), fetch_user(2));
    println!("{a:?} {b:?}");
}`;

## the ideas that make Rust click

<div class="concept-grid">
  <div class="concept">
    <div class="concept-text">
      <div class="concept-head"><span class="ordinal">01</span><h3 class="concept-title">ownership/</h3></div>
      <p>Rust's defining idea: every value has a single owner, and the compiler tracks it at build time. No garbage collector, no use-after-free — the TS habit of sharing references everywhere becomes explicit moves and borrows.</p>
      <div class="tags"><span class="tag">[move]</span><span class="tag">[borrow]</span><span class="tag">[lifetimes]</span><span class="tag">[no_gc]</span></div>
    </div>
    <div class="code-window">
      <div class="cw-bar"><span class="cw-dots"></span><span class="cw-title">ownership.rs</span></div>
      <Code code={ownCode} lang="rust" frame="none" />
    </div>
  </div>

  <div class="concept concept--reverse">
    <div class="concept-text">
      <div class="concept-head"><span class="ordinal">02</span><h3 class="concept-title">error_handling/</h3></div>
      <p>No <code>try/catch</code>, no <code>throw</code>. Errors are values: <code>Result&lt;T, E&gt;</code> and <code>Option&lt;T&gt;</code>, with the <code>?</code> operator to propagate them in one character. The type system makes you handle them — you can't forget.</p>
      <div class="tags"><span class="tag">[result]</span><span class="tag">[option]</span><span class="tag">[question_mark]</span><span class="tag">[no_exceptions]</span></div>
    </div>
    <div class="code-window">
      <div class="cw-bar"><span class="cw-dots"></span><span class="cw-title">errors.rs</span></div>
      <Code code={errCode} lang="rust" frame="none" />
    </div>
  </div>

  <div class="concept">
    <div class="concept-text">
      <div class="concept-head"><span class="ordinal">03</span><h3 class="concept-title">async_await/</h3></div>
      <p>The syntax you know — <code>async</code>/<code>await</code> — but powered by <code>Future</code>s and a runtime like Tokio instead of a single event loop. True multi-threaded, fearless concurrency the borrow checker guarantees is data-race free.</p>
      <div class="tags"><span class="tag">[tokio]</span><span class="tag">[futures]</span><span class="tag">[join]</span><span class="tag">[no_event_loop]</span></div>
