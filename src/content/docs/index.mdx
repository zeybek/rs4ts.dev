export const tsCode = `type User = { id: string; name: string };

async function getUser(id: string): Promise<User> {
  const response = await fetch("/api/users/" + id);
  if (!response.ok) throw new Error("request failed");
  return response.json();
}`;

export const rustCode = `struct User { id: String, name: String }

async fn get_user(id: &str) -> Result<User, ApiError> {
    let url = format!("/api/users/{id}");
    let response = reqwest::get(url).await?.error_for_status()?;
    Ok(response.json().await?)
}`;

export const ownCode = `let s = String::from("hello");
let s2 = s;            // ownership moves to s2
// println!("{s}");    // compile error: s no longer valid
println!("{s2}");      // ok: s2 owns the string`;

export const errCode = `fn parse_double(input: &str) -> Result<i32, ParseIntError> {
    let n: i32 = input.trim().parse()?;   // ? propagates the error
    Ok(n * 2)
}`;

export const asyncCode = `#[tokio::main]
async fn main() {
    // Tokio's runtime drives both futures concurrently
    let (a, b) = tokio::join!(fetch_user(1), fetch_user(2));
    println!("{a:?} {b:?}");
}`;

## three differences to understand first

<div class="concept-grid">
  <div class="concept">

      <div class="concept-head"><span class="ordinal">01</span><h3 class="concept-title">ownership_and_borrowing/</h3></div>
      <p>JavaScript relies on garbage collection. Rust uses ownership and borrowing to make data lifetime and mutation rules explicit at compile time.</p>
      <div class="tags"><span class="tag">[ownership]</span><span class="tag">[borrow]</span><span class="tag">[lifetimes]</span><span class="tag">[drop]</span></div>

  <div class="concept concept--reverse">
    <div class="concept-text">
      <div class="concept-head"><span class="ordinal">02</span><h3 class="concept-title">result_and_option/</h3></div>
      <p>For recoverable failures, Rust returns <code>Result&lt;T, E&gt;</code>; missing values are <code>Option&lt;T&gt;</code>. The <code>?</code> operator propagates errors without hiding them from the function signature.</p>
      <div class="tags"><span class="tag">[result]</span><span class="tag">[option]</span><span class="tag">[question_mark]</span><span class="tag">[error_types]</span></div>

  </div>

  <div class="concept">
    <div class="concept-text">
      <div class="concept-head"><span class="ordinal">03</span><h3 class="concept-title">async_and_concurrency/</h3></div>
      <p>The <code>async</code>/<code>await</code> syntax is familiar, but Rust futures are lazy and need an executor such as Tokio. Async coordinates waiting; CPU-bound parallelism is a separate choice.</p>
